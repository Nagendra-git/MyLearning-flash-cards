Go Control Constructs

    1. if Statement: The if statement executes a block of code only when a condition evaluates to true.
        
if condition {
    // statements
}
        
    
    2. if-else: When we have two decesion makings
    
if condition {
    // statements
}else{
    // statements
}     
    
    3. if-else if Ladder: Multiple decesion makings.
    
if condition {
    // statements
}else if condition{
    // statements
    .....
}else {
    // statements
}
    
    4. Nested if:
    
if condition {
    // outer statements
    if condition {
        // innner statements
    }
}   
    
    
    5. if with Short Statement (Go Feature):
        
if variable := expression; condition {
}
if num := 10; num%2 == 0 {
    fmt.Println("Even")
}

switch Statement


switch expression{
    case 1: 
        //statement
    .
    .
    case n :
        //statements
    default:
        //statements
}

    Multiple Values in a Case:
    
switch expression{
    case 1: 
        //statement
    case 2,3,4,5:
        //statement
}
    
    Expression-less switch: Acts like an if-else if ladder.
    
switch{
    case 1: 
        //statement
    .
    .
    case n :
        //statements
    default:
        //statements
}
    

    fallthrough: unlike java, Go does not automatically fall through to the next case.
    
switch expression{
    case 1: 
        //statement
        fallthrough
    case 2,3,4,5:
        //statement
}
    
    Type switch:
        
var value interface{} = "Go"

switch v := value.(type) {
case int:
    fmt.Println("Integer", v)
case string:
    fmt.Println("String", v)
default:
    fmt.Println("Unknown")
}

For Loop

    Go has only one looping keyword: for.
    Traditional loop:
    
for i := 1; i <= 5; i++ {
    fmt.Println(i)
}
    

    for as while:
    
i := 1
for i <= 5 {
    fmt.Println(i)
    i++
}
    

    Infinite Loop: Use break to exit.
    
for {
    fmt.Println("Running...")
}
    

    range Loop: range iterates over collections.
    
nums := []int{10, 20, 30}

for index, value := range nums {
    fmt.Println(index, value)
}
    

    Ignore Index: The blank identifier (_) ignores the index.
    
for _, value := range nums {
    fmt.Println(value)
}
    

    Map:
    
student := map[string]int{
    "John": 90,
    "Alice": 95,
}

for key, value := range student {
    fmt.Println(key, value)
}
    
    Note: Map iteration order in Go is not guaranteed. Don't rely on it being the same each run.

    String:
    
name := "Go"

for index, ch := range name {
    fmt.Println(index, string(ch))
}
    

    break : it will brake the Loop
    continue: it will skip the current itration.
    Labels: Useful when breaking out of nested loops.
    
Outer:
for i := 1; i <= 3; i++ {
    for j := 1; j <= 3; j++ {
        if i == 2 && j == 2 {
            break Outer
        }
        fmt.Println(i, j)
    }
}
    

    goto : (Rarely Used)
        Avoid goto unless there's a compelling reason; it can make code harder to follow.
    
i := 1

Loop:
fmt.Println(i)

i++

if i <= 5 {
    goto Loop
}
