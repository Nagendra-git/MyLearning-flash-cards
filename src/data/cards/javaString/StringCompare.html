== vs equals()

    == -> Checks Reference
    String s1 = "java";
    String s2 = "java";
    sysout(s1==s2); //true

    equals() -> Checks Content.
    String s1 = new String("java");
    String s2 = new String("java");
    sysout(s1.equals(s2));//true (content is same)
    sysout(s1 == s2 ); //false (Diffrent objects)
